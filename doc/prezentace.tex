\begin{frame}
    \center
  \frametitle{CIE RGB $\rightarrow$ CIE rgb}
  Nezajímá nás jas, ale jen chromaticita $\rightarrow$ normalizujeme.

  \begin{align}
    \begin{split}
    r &= R / (R + G + B)\\
    g &= G / (R + G + B)\\
    b &= B / (R + G + B)
  \end{split}
    \label{eq:rgb}
  \end{align}

  \begin{align}
    r+g+b=1
  \end{align}

  Souřadnici \textit{b} nepotřebujeme, protože jde dopočítat.
\end{frame}
